cflow的典型配置

      forwarding-options {

          sampling {

              input {

                  family inet {

                      rate 1;

                                    }

                       }

              output {

                  cflowd 10.42.69.10 {

                      port 2055;

                      version 5;

                                                }

                       }

                         }

                                   }

 

      firewall {

          filter Cflowfilter {

              term all {

                  then {

                      sample;

                      accept;

                  }

              }

          }

      }

 

interfaces {

          ge-0/3/0 {

              unit 0 {

                  family inet {

                      filter {

                          input Cflowfilter;

                                }

                      address 10.42.69.1/24;

                                     }

                       }

               }

            }
